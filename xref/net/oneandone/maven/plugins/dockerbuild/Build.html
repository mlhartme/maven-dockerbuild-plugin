54  /**
55   * Builds a Docker image for this Maven module.
56   */
57  @Mojo(name = "build", defaultPhase = LifecyclePhase.PACKAGE, requiresDependencyResolution = ResolutionScope.NONE, threadSafe = true)
58  public class Build extends Base {
59      /** The maven groupId that contains the available docker builds */
60      @Parameter(property = "dockerbuild.library", defaultValue = "com.dockerbuild.library")
61      private final String library;
62  
63      /** Name of the dockerbuild to use. */
64      @Parameter(required = true)
65      private final String dockerbuild;
66  
67      /** Version of the dockerbuild. If specified, this has to be the latest version. Otherwise, the latest version is used automatically. */
68      @Parameter()
69      private final String version;
70  
71      /** Don't use Docker build cache */
72      @Parameter(property = "dockerbuild.noCache", defaultValue = "false")
73      private final boolean noCache;
74  
75      /** Image tag created by this build. Provides placeholders inspired by https://maven.fabric8.io/#image-configuration */
76      @Parameter(property = "dockerbuild.image", defaultValue = "%g/%a:%V")
77      private final String image;
78  
79      /** Comment passed to the buildComment argument. */
80      @Parameter(property = "dockerbuild.comment", defaultValue = "")
81      private final String comment;
82  
83      /** Explicit argument values passed to the build. */
84      @Parameter
85      private Map<String, String> arguments;
86  
87      /** Used internally */
88      @Parameter(property = "project", required = true, readonly = true)
89      private final MavenProject project;
90  
91      //--
92  
93      @Component
94      private RepositorySystem repoSystem;
95  
96      /** Used internally */
97      @Parameter(defaultValue = "${repositorySystemSession}", readonly = true)
98      private RepositorySystemSession repoSession;
99  
100     /** Used internally */
101     @Parameter(defaultValue = "${project.remoteProjectRepositories}", readonly = true)
102     private List<RemoteRepository> remoteRepos;
103 
104     public Build() throws IOException {
105         this.library = null;
106         this.dockerbuild = null;
107         this.version = null;
108         this.noCache = false;
109         this.image = "";
110         this.comment = "";
111         this.arguments = new HashMap<>();
112         this.project = null;
113     }
114 
115     @Override
116     public void doExecute(DockerClient docker) throws IOException, MojoExecutionException {
117         Log log;
118         String repositoryTag;
119         FileNode contextDir;
120         Context context;
121         long started;
122         FileNode jar;
123         Map<String, String> actuals;
124         String id;
125         BuildImageCmd build;
126         FileNode buildLog;
127 
128         log = getLog();
129         repositoryTag = new Placeholders(project).resolve(image);
130         jar = resolveDockerbuild();
131         contextDir = context();
132         log.info("rm -rf " + contextDir + "; mkdir " + contextDir);
133         log.info("(cd " + contextDir + " && jar xf " + jar + ")");
134 
135         context = Context.create(jar, dockerbuild, contextDir);
136         buildLog = buildLog();
137         buildLog.getParent().mkdirsOpt();
138         started = System.currentTimeMillis();
139         imageFile().getParent().mkdirsOpt();
140         imageFile().writeString(repositoryTag);
141         Arguments a = context.arguments(log);
142         a.addArtifacts(context, world.file(project.getBuild().getDirectory()), project.getBuild().getFinalName());
143         a.addBuild(comment);
144         a.addPom(project);
145         a.addExplicit(arguments);
146         actuals = a.result();
147         try (InputStream tarSrc = context.tar().newInputStream()) {
148             build = docker.buildImageCmd()
149                     .withTarInputStream(tarSrc)
150                     .withNoCache(noCache)
151                     .withTags(Collections.singleton(repositoryTag));
152             for (Map.Entry<String, String> entry : actuals.entrySet()) {
153                 build.withBuildArg(entry.getKey(), entry.getValue());
154             }
155             log.info(cli(build, contextDir) + " >" + buildLog);
156             try (PrintWriter logfile = new PrintWriter(buildLog.newWriter())) {
157                 id = build.exec(new BuildListener(log, logfile)).awaitImageId();
158             }
159         } catch (MojoExecutionException e) {
160             log.error("build failed");
161             for (String line : buildLog.readLines()) {
162                 log.error("  " + line);
163             }
164             throw e;
165         }
166         log.info("Done: " + repositoryTag);
167         log.debug("id=" + id + " seconds=" + (System.currentTimeMillis() - started) / 1000);
168     }
169 
170     /** command-line equivalent of the rest call we're using */
171     private static String cli(BuildImageCmd cmd, FileNode context) {
172         StringBuilder cli;
173 
174         cli = new StringBuilder("docker build -t \"" + cmd.getTags().iterator().next() + '"');
175         if (cmd.hasNoCacheEnabled()) {
176             cli.append(" --no-cache");
177         }
178         cli.append(" \\\n");
179         for (Map.Entry<String, String> entry : cmd.getBuildArgs().entrySet()) {
180             cli.append("           --build-arg ");
181             cli.append(entry.getKey());
182             cli.append('=');
183             cli.append(entry.getValue());
184             cli.append(" \\\n");
185         }
186         cli.append(" " + context);
187         return cli.toString();
188     }

190     //-- artifact resolution, see https://maven.apache.org/resolver/maven-resolver-demos/maven-resolver-demo-maven-plugin/xref/index.html
191 
192     private FileNode resolveDockerbuild() throws MojoExecutionException {
193         String gav;
194         Artifact artifact;
195         ArtifactRequest request;
196         ArtifactResult result;
197         FileNode file;
198 
199         gav = library + ":" + dockerbuild + ":" + checkedVersion();
200         getLog().info("resolve " + gav);
201         try {
202             artifact = new DefaultArtifact(gav);
203         } catch (IllegalArgumentException e) {
204             throw new MojoExecutionException("invalid dockerbuild gav: " + gav + ": " + e.getMessage(), e);
205         }
206         request = new ArtifactRequest();
207         request.setArtifact(artifact);
208         request.setRepositories(remoteRepos);
209 
210         try {
211             result = repoSystem.resolveArtifact(repoSession, request);
212         } catch (ArtifactResolutionException e) {
213             throw new MojoExecutionException(dockerbuild + ": failed to resolve dockerbuild: " + e.getMessage(), e);
214         }
215         file = world.file(result.getArtifact().getFile());
216         return file;
217     }
218 
219     private String checkedVersion() throws MojoExecutionException {
220         String gav;
221         Artifact artifact;
222         VersionRangeRequest request;
223         VersionRangeResult result;
224 
225         gav = library + ":" + dockerbuild + ":[" + (version == null ? "0" : version) + ",)";
226         try {
227             artifact = new DefaultArtifact(gav);
228         } catch (IllegalArgumentException e) {
229             throw new MojoExecutionException("invalid dockerbuild version gav: " + gav + ": " + e.getMessage(), e);
230         }
231         request = new VersionRangeRequest();
232         request.setArtifact(artifact);
233         request.setRepositories(remoteRepos);
234         try {
235             result = repoSystem.resolveVersionRange(repoSession, request);
236         } catch (VersionRangeResolutionException e) {
237             throw new MojoExecutionException("dockerbuild version check failed: " + gav + ": " + e.getMessage(), e);
238         }
239         getLog().debug("available versions: " + result.getVersions());
240         if (version == null) {
241             return result.getHighestVersion().toString();
242         } else {
243             if (result.getHighestVersion() == null) {
244                 throw new MojoExecutionException(gav + ": no versions available");
245             }
246             if (!version.equals(result.getHighestVersion().toString())) {
247                 throw new MojoExecutionException("newer version(s) available: " + result);
248             }
249             return version;
250         }
251     }
